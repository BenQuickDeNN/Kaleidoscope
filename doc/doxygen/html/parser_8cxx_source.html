146 static std::unique_ptr<ExprAST> ParseBinOpRHS(int ExprPrec,
147  std::unique_ptr<ExprAST> LHS)
148 {
149  // If this is a binop, find its precedence.
150  while (true)
151  {
152  int TokPrec = GetTokPrecedence();
153 
154  // If this is a binop that binds at least as tightly as the current binop
155  // consume it, otherwise we are done
156  if (TokPrec < ExprPrec)
157  return LHS;
158 
159  // well this is a binop
160  int BinOp = CurTok;
161  getNextToken(); // eat binop
162 
163  // Parse the primary expression after the binary operator
164  auto RHS = ParsePrimary();
165  if (!RHS)
166  return nullptr;
167 
168  // 如果下一个token的优先级比当前token的优先级更高，那么就把RHS当做LHS
169  // 优先用右边两个token构造语法树节点
170  int NextPrec = GetTokPrecedence();
171  if (TokPrec < NextPrec)
172  {
173  RHS = ParseBinOpRHS(TokPrec + 1, std::move(RHS));
174  if (!RHS)
175  return nullptr;
176  }
177 
178  // Merge LHS/RHS, generate AST node
179  LHS = std::make_unique<BinaryExprAST>(BinOp, std::move(LHS),
180  std::move(RHS));
181  }
182 }
